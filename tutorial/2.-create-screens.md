# 2. Create Screens


### Create Screens

1. create file like Home.js 
2. add your file in createPlugin that make it register for navigation process
3. example code
```jsx
import React from 'react';
import { BlueBaseApp, createPlugin, BlueBase } from '@bluebase/core';
import LoginScreen from './components/Login';
import Home from './components/Home';
import BlueBasePluginReactRouter from '@bluebase/plugin-react-router';
import BlueBasePluginSettingsApp from '@bluebase/plugin-settings-app';
import BlueBasePluginMaterialUI from '@bluebase/plugin-material-ui';
import BlueBasePluginResponsiveGrid from '@bluebase/plugin-responsive-grid';
import BlueBasePluginLauncher from '@bluebase/plugin-launcher';
import BlueBasePluginJsonSchemaComponents from '@bluebase/plugin-json-schema-components';
import { MaterialCommunityIcons } from '@bluebase/plugin-vector-icons';

export const createPlugins = createPlugin({
  key: "rootUrl",
  components: {
    Home: Home,
    HomeScreen: Home, 
    LoginScreen: LoginScreen
  },
  routes: [
    {
      name: 'Home',
      path: 'Home',
      screen: Home
    },
    {
      name: 'LoginScreen',
      path: 'LoginScreen',
      screen: LoginScreen
    }
  ]
});
export default function App() {
  return (
    <BlueBaseApp
      plugins={[
        BlueBasePluginMaterialUI,
        BlueBasePluginResponsiveGrid,
        BlueBasePluginReactRouter,
        MaterialCommunityIcons,
        BlueBasePluginLauncher,
        BlueBasePluginJsonSchemaComponents,
        BlueBasePluginSettingsApp,
        createPlugins
      ]}

    />
  );
}
```
4. now you can easily navigate your screen by using useNavigation
```jsx
 import { useNavigation } from '@bluebase/core';
	const  {navigate}  = useNavigation()
     <TouchableOpacity onPress={() => navigate('Home')}>
             <Text style={style.colorWhite}>GO TO HOME SCREEN </Text>
     </TouchableOpacity>

```
