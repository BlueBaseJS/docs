	key: 'tasks',
	name: 'Tasks',
	description: 'A todo app made with BlueBase framework.',

	components: {
		// Components
		ToDoAppIcon,

		// Screens
		CompletedTasksListScreen,
		PendingTasksListScreen,
	},

	icon: {
		component: 'ToDoAppIcon',
		type: 'component',
	},

	indexRoute: 'TasksApp',

	routes: [{
		name: 'TasksApp',
		path: '',
		screen: Noop,

			title: 'My Tasks',

		navigator: {
