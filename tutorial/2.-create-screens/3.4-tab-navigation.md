# 3.4 Tab Navigation





```
import { Noop, RouteConfig } from '@bluebase/components';
import { RouteOptions } from '@bluebase/core';
import React from 'react';

import TaskCreateButton from './components/TaskCreateButton';

export const routes = async (_options: RouteOptions): Promise<RouteConfig[]> => {

	return [{
		name: 'CreateTask',
		screen: 'CreateTaskScreen',
		path: 'create',
		exact: true,

		options: {
			title: 'Create Task',
		},
	},
	{
		name: 'EditTask',
		screen: 'EditTaskScreen',
		path: 't/:taskId',
		exact: true,

		options: {
			title: 'Edit Task',
		},
	},
	{
		name: 'TasksApp',
		screen: Noop,
		path: '',
		exact: false,

		options: {
			title: 'My Tasks',
			headerRight: () => <TaskCreateButton />
		},

		navigator: {
			headerMode: 'none',
			type: 'tab',
			routes: [
				{
					exact: true,
					name: 'PendingTask',
					path: 'pending',
					screen: 'PendingTasksScreen',

					options: {
						title: 'Pending',
					},
				},
				{
					exact: true,
					name: 'CompletedTasks',
					path: 'completed',
					screen: 'CompletedTasksScreen',

					options: {
						title: 'Completed',
					},
				},
			],
		},
	}];
};

```

Icons



```
import { Icon, Noop, RouteConfig } from '@bluebase/components';
import { RouteOptions } from '@bluebase/core';
import React from 'react';

import TaskCreateButton from './components/TaskCreateButton';

export const routes = async (_options: RouteOptions): Promise<RouteConfig[]> => {

	return [{
		name: 'CreateTask',
		screen: 'CreateTaskScreen',
		path: 'create',
		exact: true,

		options: {
			title: 'Create Task',
		},
	},
	{
		name: 'EditTask',
		screen: 'EditTaskScreen',
		path: 't/:taskId',
		exact: true,

		options: {
			title: 'Edit Task',
		},
	},
	{
		name: 'TasksApp',
		screen: Noop,
		path: '',
		exact: false,

		options: {
			title: 'My Tasks',
			headerRight: () => <TaskCreateButton />
		},

		navigator: {
			headerMode: 'none',
			type: 'tab',

			routes: [
				{
					exact: true,
					name: 'PendingTask',
					path: 'pending',
					screen: 'PendingTasksScreen',

					options: {
						title: 'Pending',
						tabBarIcon: ({ color }: any) => <Icon name="checkbox-multiple-blank-outline" color={color} />,
					},
				},
				{
					exact: true,
					name: 'CompletedTasks',
					path: 'completed',
					screen: 'CompletedTasksScreen',

					options: {
						title: 'Completed',
						tabBarIcon: ({ color }: any) => <Icon name="checkbox-multiple-marked" color={color} />,
					},
				},
			],

			tabBarOptions: {
				showIcon: true,
				tabStyle: {
					flexDirection: 'row',
				},
			},
		},
	}];
};
```
