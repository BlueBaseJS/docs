	key: 'tasks',
	name: 'Tasks',
	description: 'A todo app made with BlueBase framework.',

	components: {
		// Components
		ToDoAppIcon,

		// Screens
		...screens,
	},

	icon: {
		component: 'ToDoAppIcon',
		type: 'component',
	},

	indexRoute: 'TasksApp',

	routes: [{
		name: 'TasksApp',
		path: '',
		screen: 'PendingTasksScreen',
		exact: false,

		options: {
			title: 'My Tasks',
		},
	}]
});
